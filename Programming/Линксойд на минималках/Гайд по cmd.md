```комманды
mkdir папка - для создания
mkdir моя/ папка - экранирование символов
cd .. - назад
touch - для создания файлов
nano - для открытия файлов
echo "" > файл - для перенаправления вывода в консоль в файл
cat - для вывода содержимого
> - перезапись (write)
>> - добавление (append)
rm - удаление файлов
rm -r - для удаления папки и файлов в ней
cmod - изм. права на файл (7-общие7-кто создал7-остальные)
cp <файл> <новыыйфайл> - копирование
mv <файл> <новое название> - для переименования


```
